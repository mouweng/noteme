# 常用模板

## gcd & lcm

```java
int gcd(int a, int b){
	return b == 0 ? a : gcd(b, a % b);
}

int lcm(int a, int b) {
    return a * b / gcd(a, b);
}
```

## 素数

```java
boolean isPrime(int x) {
    for (int i = 2; i * i <= x; ++i) {
        if (x % i == 0) return false;
    }
    return true;
}
```

## 并查集

```java
class Main {
    static class UnionFind {
        public int[] f;
        public UnionFind(int n) {
            f = new int[n + 1];
            for (int i = 1; i <= n; i ++) f[i] = i;
        }
        public int find(int x) {
            return f[x] == x ? x : (f[x] = find(f[x]));
        }
        public void union(int a, int b) {
            int x = find(a), y = find(b);
            if (x != y) f[y] = x;
        }
        public int size() {
            HashSet<Integer> s = new HashSet<>();
            for (int i = 1; i < f.length; i ++) s.add(find(i));
            return s.size();
        }
    }
    public static void main(String[] args) {
        UnionFind uf = new UnionFind(10);
        uf.union(1, 2);uf.union(2, 3);
        System.out.println(uf.find(1) == uf.find(3));
        System.out.println(uf.size());
    }
}
```

## 快速排序

```java
class Solution {
    // 返回数组中第k大的元素
    public int findKthLargest(int[] nums, int k) {
        int len = nums.length;
        quickSort(nums, 0 , len - 1);
        return nums[len - k];
    }
    public void quickSort(int[] nums, int l, int r) {
        if (l >= r) return;
        int q = nums[l];
        int i = l, j = r;	
        while (i < j) {
            while (i < j && nums[j] >= q) j --;
            nums[i] = nums[j];
            while (i < j && nums[i] <= q) i ++;
            nums[j] = nums[i];
        }
        nums[i] = q;
        quickSort(nums, l, i - 1);
        quickSort(nums, i + 1, r);
    }
}
```

## 归并排序

```java
class Solution {
    // 返回数组中第k大的元素
    public int findKthLargest(int[] nums, int k) {
        int len = nums.length;
        merge(nums, 0 , len - 1);
        return nums[len - k];
    }
    public void merge(int[] nums, int l, int r) {
        if (l >= r) return;
        int mid = l + r >> 1;
        merge(nums, l , mid);
        merge(nums, mid + 1, r);

        // 合并l-mid 和 mid+1 - r部分 
        int[] temp = new int[r - l + 1];
        int i = l, j = mid + 1, t = 0;
        while (i <= mid && j <= r) {
            if (nums[i] <= nums[j]) temp[t ++] = nums[i ++];
            else temp[t ++] = nums[j ++];
        }

        // 剩余部分处理
        if (i > mid) while(j <= r) temp[t ++] = nums[j ++];
        else while(i <= mid) temp[t ++] = nums[i ++];

        // 把temp数组赋值到nums数组上
        for (int k = 0; k < temp.length; k ++) {
            nums[l ++] = temp[k];
        }
    }
}
```

## 堆排序

```java
class Solution {
    // 返回数组中第k大的元素
    public int findKthLargest(int[] nums, int k) {
        int len = nums.length;
        heapSort(nums);
        return nums[len - k];
    }
    
    
    // 交换
    void swap(int[] nums, int i, int k) {
        int temp = nums[i];
        nums[i] = nums[k];
        nums[k] = temp;
    }

    // 调整
    void shift(int[] nums, int k, int len) {
        for (int i = k * 2 + 1; i < len; i = i * 2 + 1) {
            if(i + 1 < len && nums[i + 1] > nums[i]) i ++;
            if (nums[i] > nums[k])
                swap(nums, i, k);
            k = i;
        }
    }

    // 建堆
    void buildHeap(int[] nums, int len) {
        for (int i = len / 2 - 1; i >= 0; i --) {
            shift(nums, i, len);
        }
    }
    // 堆排序
    void heapSort(int[] nums) {
        for (int i = 0; i < nums.length; i ++) {
            buildHeap(nums, nums.length - i);
            swap(nums, 0, nums.length - 1 - i);
        }
    }
}
```
